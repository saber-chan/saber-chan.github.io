python-turtle

2019-08-20 · 62 个字 · 1 分钟阅读

使用嵌套循环画出一朵花。

import turtle
links = [1, 2, 3, 4, 5, 6]
sides = [1, 2, 3, 4, 5, 6]

weaver = turtle.Turtle()
weaver.width(5)
weaver.color('orange')

# Move back so the chain is centered.
weaver.penup()
weaver.back(80)
weaver.pendown()

for link in links:
    # Draw a hexagon.
    for side in sides:
        weaver.forward(50)
        weaver.right(60)

    # Scoot over to the next link.
    weaver.penup()
    weaver.forward(50)
    weaver.left(60)
    weaver.forward(-37)
    weaver.pendown()

weaver.hideturtle()
